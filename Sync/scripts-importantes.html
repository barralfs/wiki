Scripts importantes

  • Tabela com resumo dos documentos fiscais
SELECT * FROM NOTA_FISCAL_DESTINADA
  • Tabela com todas as informações dos documentos fiscais
SELECT * FROM NOTA_FISCAL
  • Script para excluir Resumo NFSe NAO IDENTIFICADO ou AGUARDANDO DISPONIBILIDADE
SELECT * FROM NOTA_FISCAL_DESTINADA
WHERE MODELO = 98
AND RAZAO_SOCIAL = 'NAO IDENTIFICADO'
DELETE FROM NOTA_FISCAL_DESTINADA
WHERE MODELO = 98
AND RAZAO_SOCIAL = 'NAO IDENTIFICADO'

 

SELECT * FROM NOTA_FISCAL_DESTINADA
WHERE MODELO = 98
AND RAZAO_SOCIAL LIKE '%AGUARDANDO DISPONIBILIDADE%'
DELETE FROM NOTA_FISCAL_DESTINADA
WHERE MODELO = 98
AND RAZAO_SOCIAL LIKE '%AGUARDANDO DISPONIBILIDADE%'
  • Script para excluir Documento Completo NFSe NAO IDENTIFICADO ou AGUARDANDO DISPONIBILIDADE
SELECT * FROM NOTA_FISCAL
WHERE MODELO = 98
AND NUMERO = ''
AND CHAVE_ACESSO = ''
DELETE NOTA_FISCAL
WHERE MODELO = 98
AND NUMERO = ''
AND CHAVE_ACESSO = ''
  • Resetar NSU de NFSe
SELECT * FROM CONFIG
WHERE NOME LIKE '%NSU_NFSE%'
UPDATE CONFIG SET VALOR = 0
WHERE NOME LIKE '%NSU_NFSE%'
  • Apagar Schema_History
SELECT * FROM SCHEMA_HISTORY
ORDER BY VERSAO
DELETE FROM SCHEMA_HISTORY
WHERE VERSAO >= 30
  • Apagar número da versão atual do Sync, para permitir instalar versões anteriores
DELETE FROM ARQUIVO
