Fórmulas de Tributos

Este artigo apresenta as fórmulas padrão para cadastro de tributos no Sync, para NF-e e NFS-e, e a ordem correta de preenchimento.

O local de cadastro e edição dos tributos está descrito no artigo Parametrização de tributos.

As fórmulas devem ser cadastradas sempre nesta sequência:

  1. Base Cálculo
  2. Tributo Base
  3. Alíquota

Fórmulas – NF-e

ICMS

ISNULL(ITEM.ICMS_BASE_CALC,0)

ISNULL(ITEM.ICMS_VALOR,0)

ISNULL(ITEM.ICMS_ALIQUOTA,0)

ICMS ST

ISNULL(ITEM.ICMS_ST_BASE_CALC,0)

ISNULL(ITEM.ICMS_ST_VALOR,0)

ISNULL(ITEM.ICMS_ST_ALIQUOTA,0)

IPI

ISNULL(ITEM.IPI_BASE_CALC,0)

ISNULL(ITEM.IPI_VALOR,0)

ISNULL(ITEM.IPI_ALIQUOTA,0)

PIS

ISNULL(ITEM.PIS_BASE_CALC,0)

ISNULL(ITEM.PIS_VALOR,0)

ISNULL(ITEM.PIS_ALIQUOTA,0)

COFINS

ISNULL(ITEM.COFINS_BASE_CALC,0)

ISNULL(ITEM.COFINS_VALOR,0)

ISNULL(ITEM.COFINS_ALIQUOTA,0)

ICMS FCP

ISNULL(ITEM.ICMS_FCP_BASE_CALC,0)

ISNULL(ITEM.ICMS_FCP_VALOR,0)

ISNULL(ITEM.ICMS_FCP_ALIQUOTA,0)

ICMS ST FCP

ISNULL(ITEM.ICMS_ST_FCP_BASE_CALC,0)

ISNULL(ITEM.ICMS_ST_FCP_VALOR,0)

ISNULL(ITEM.ICMS_ST_FCP_ALIQUOTA,0)

Fórmulas – NFS-e

INSS

ISNULL(NOTA.TOTAL_SERVICOS,0)

ISNULL(NOTA.TOTAL_RETENCAO_CP,0)

ISNULL((ISNULL(NOTA.TOTAL_RETENCAO_CP,0)/NULLIF(NOTA.TOTAL_SERVICOS,0)*100),0)

ISS Retido

ISNULL(ITEM.ISSQN_BASE_CALC,0)

CASE
WHEN ISNULL(ITEM.ISSQN_VALOR_RETIDO,0) > 0
THEN ISNULL(ITEM.ISSQN_VALOR_RETIDO,0)
ELSE 0
END

CASE
WHEN ISNULL(ITEM.ISSQN_VALOR_RETIDO,0) > 0
THEN ISNULL(ITEM.ISSQN_ALIQUOTA,0)
ELSE 0
END

PIS, COFINS e CSLL Retidos

ISNULL(NOTA.TOTAL_SERVICOS,0)

CASE
WHEN NOTA.TOTAL_PIS > 0 AND NOTA.TOTAL_COFINS > 0 AND NOTA.TOTAL_RETENCAO_CSLL > 0
THEN ISNULL((NOTA.TOTAL_PIS + NOTA.TOTAL_COFINS + NOTA.TOTAL_RETENCAO_CSLL),0)
ELSE 0
END

ISNULL('4,65',0)

IRRF

ISNULL(NOTA.TOTAL_SERVICOS,0)

ISNULL(NOTA.TOTAL_RETENCAO_IRRF,0)

ISNULL((ISNULL(NOTA.TOTAL_RETENCAO_IRRF,0)/NULLIF(NOTA.TOTAL_SERVICOS,0)*100),0)

PIS

ISNULL(NOTA.TOTAL_SERVICOS,0)

ISNULL(NOTA.TOTAL_PIS,0)

ISNULL((ISNULL(NOTA.TOTAL_PIS,0)/NULLIF(NOTA.TOTAL_SERVICOS,0)*100),0)

COFINS

ISNULL(NOTA.TOTAL_SERVICOS,0)

ISNULL(NOTA.TOTAL_COFINS,0)

ISNULL((ISNULL(NOTA.TOTAL_COFINS,0)/NULLIF(NOTA.TOTAL_SERVICOS,0)*100),0)
